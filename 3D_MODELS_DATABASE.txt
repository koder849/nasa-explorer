# 5. GLB files are optimized for web viewing

## EXAMPLE USAGE IN CODE:

import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader.js';

const models = [
  {
    name: '70-Meter Dish',
    url: 'https://raw.githubusercontent.com/nasa/NASA-3D-Resources/master/3D%20Models/70-meter%20Dish/70%20meter%20dish.glb',
    category: 'Antennas'
  },
  {
    name: 'Hubble Space Telescope',
    url: 'https://raw.githubusercontent.com/nasa/NASA-3D-Resources/master/3D%20Models/Hubble%20Space%20Telescope%20(A)/Hubble%20Space%20Telescope%20(A).glb',
    category: 'Telescopes'
  },
  // ... more models
];

const loader = new GLTFLoader();
loader.load(models[0].url, (gltf) => {
  scene.add(gltf.scene);
});
